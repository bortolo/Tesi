\frametitle{Setting}
\begin{center}
\begin{equation*}
\left\{ \begin{array}{l c} 
\nabla \cdot ( -\epsilon_0 \epsilon_r \nabla \varphi) = q(p-n+D) & \text{in $\Omega_{Si}$}\\
\nabla \varphi \cdot \vect{n} =0 & \text{su $\Gamma_{Si}$ }\\
\varphi = \varphi_K & \text{su $\Gamma_{K}$}\\
\varphi = \varphi_A & \text{su $\Gamma_{A}$}\\
\end{array} \right.
\end{equation*}
\end{center}
